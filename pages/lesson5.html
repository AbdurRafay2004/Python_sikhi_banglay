পাঠ ৫: লিস্ট, টাপল, এবং ডিকশনারি

লিস্ট (List)

লিস্ট হল Python-এর একটি অত্যন্ত শক্তিশালী ডেটা স্ট্রাকচার যা বিভিন্ন ধরনের এলিমেন্ট সংরক্ষণ করতে পারে। লিস্ট পরিবর্তনযোগ্য (mutable) অর্থাৎ তৈরির পরে আমরা এর মান পরিবর্তন করতে পারি।

# লিস্ট তৈরি
ফল = ["আম", "কলা", "কমলা", "আনারস"]

# লিস্ট এর এলিমেন্ট অ্যাক্সেস করা
print(ফল[0])  # আউটপুট: আম

# লিস্ট এর এলিমেন্ট পরিবর্তন করা
ফল[1] = "লিচু"
print(ফল)  # আউটপুট: ["আম", "লিচু", "কমলা", "আনারস"]

# লিস্ট এ এলিমেন্ট যোগ করা
ফল.append("পেয়ারা")
print(ফল)  # আউটপুট: ["আম", "লিচু", "কমলা", "আনারস", "পেয়ারা"]

# লিস্ট থেকে এলিমেন্ট মুছে ফেলা
ফল.remove("কমলা")
print(ফল)  # আউটপুট: ["আম", "লিচু", "আনারস", "পেয়ারা"]

টাপল (Tuple)

টাপল লিস্টের মতই, কিন্তু এটি অপরিবর্তনযোগ্য (immutable)। একবার তৈরি করার পরে টাপলের এলিমেন্ট পরিবর্তন করা যায় না।

# টাপল তৈরি
রঙ = ("লাল", "সবুজ", "নীল")

# টাপল এর এলিমেন্ট অ্যাক্সেস করা
print(রঙ[1])  # আউটপুট: সবুজ

# টাপল এর এলিমেন্ট পরিবর্তন করা যাবে না
# রঙ[0] = "হলুদ"  # এটি এরর দেখাবে

# টাপলের দৈর্ঘ্য পাওয়া
print(len(রঙ))  # আউটপুট: 3

# টাপল আনপ্যাকিং
র1, র2, র3 = রঙ
print(র1, র2, র3)  # আউটপুট: লাল সবুজ নীল

ডিকশনারি (Dictionary)

ডিকশনারি হলো key-value জোড়ায় ডেটা সংরক্ষণ করার একটি উপায়। এটি খুব দ্রুত ডেটা খুঁজে বের করতে সাহায্য করে।

# ডিকশনারি তৈরি
ছাত্র = {
    "নাম": "করিম",
    "বয়স": 20,
    "শ্রেণী": "বিএসসি",
    "বিষয়": ["গণিত", "পদার্থবিজ্ঞান", "রসায়ন"]
}

# ডিকশনারি থেকে ভ্যালু অ্যাক্সেস করা
print(ছাত্র["নাম"])  # আউটপুট: করিম

# .get() মেথড ব্যবহার করা (ভ্যালু না থাকলে এরর দেখাবে না)
print(ছাত্র.get("ঠিকানা", "পাওয়া যায়নি"))  # আউটপুট: পাওয়া যায়নি

# ডিকশনারিতে নতুন key-value যোগ করা
ছাত্র["ঠিকানা"] = "ঢাকা"
print(ছাত্র)

# ডিকশনারির সব কী দেখা
print(ছাত্র.keys())  # আউটপুট: dict_keys(['নাম', 'বয়স', 'শ্রেণী', 'বিষয়', 'ঠিকানা'])

# ডিকশনারির সব ভ্যালু দেখা
print(ছাত্র.values())

বাস্তব উদাহরণ

একটি ছোট ফোন বুক অ্যাপ্লিকেশন তৈরি করা যাক:

# ফোন বুক অ্যাপ্লিকেশন
ফোন_বুক = {}

# কন্টাক্ট যোগ করার ফাংশন
def কন্টাক্ট_যোগ_করো(নাম, নম্বর):
    ফোন_বুক[নাম] = নম্বর
    print(f"{নাম} এর নম্বর সংরক্ষণ করা হয়েছে।")

# কন্টাক্ট খোঁজার ফাংশন
def কন্টাক্ট_খোঁজো(নাম):
    if নাম in ফোন_বুক:
        return f"{নাম} এর নম্বর: {ফোন_বুক[নাম]}"
    else:
        return f"{নাম} কন্টাক্ট লিস্টে নেই।"

# কন্টাক্ট আপডেট করার ফাংশন
def কন্টাক্ট_আপডেট(নাম, নতুন_নম্বর):
    if নাম in ফোন_বুক:
        ফোন_বুক[নাম] = নতুন_নম্বর
        return f"{নাম} এর নম্বর আপডেট করা হয়েছে।"
    else:
        return f"{নাম} কন্টাক্ট লিস্টে নেই।"

# কয়েকটি কন্টাক্ট যোগ করি
কন্টাক্ট_যোগ_করো("আমিন", "01712345678")
কন্টাক্ট_যোগ_করো("করিম", "01898765432")

# কন্টাক্ট খোঁজা
print(কন্টাক্ট_খোঁজো("আমিন"))  # আউটপুট: আমিন এর নম্বর: 01712345678

# কন্টাক্ট আপডেট
print(কন্টাক্ট_আপডেট("করিম", "01654321789"))

# সব কন্টাক্ট দেখা
print("ফোন বুক:")
for নাম, নম্বর in ফোন_বুক.items():
    print(f"{নাম}: {নম্বর}")

অনুশীলন সমস্যা

নিচের সমস্যাগুলি সমাধান করার চেষ্টা করুন:

  1. একটি শপিং লিস্ট অ্যাপ্লিকেশন তৈরি করুন যেখানে আপনি আইটেম যোগ, মুছে ফেলা, এবং দেখতে পারবেন।
  2. একটি ছাত্রদের স্কোর ট্র্যাকিং সিস্টেম তৈরি করুন যেখানে ছাত্রের নাম (key) এবং তাদের স্কোর (value) থাকবে। তারপর, সর্বোচ্চ স্কোর, সর্বনিম্ন স্কোর, এবং গড় স্কোর বের করুন।
